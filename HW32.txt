1. Получить список всех сотрудников из 50го отдела (department_id) с
зарплатой(salary), большей 4000.

select employee_id, first_name, last_name from employees
	where department_id = 50 and salary > 4000;

2. Получить список всех сотрудников у которых последняя буква в имени равна 'a'.

select * from employees
	where first_name like '%a';

3. Получить список всех сотрудников из 50го и из 80го отдела (department_id) у
которых есть бонус (значение в колонке commission_pct не пустое).

select * from employees
	where commission_pct is not null and department_id in (50, 80);

4. Получить список всех сотрудников у которых зарплата находится в промежутке от 8000 до 9000 (включительно).

select * from employees
	where salary between 8000 and 9000;

5. Получить список всех сотрудников заменив в значении PHONE_NUMBER все '.' на '-'.

select employee_id, first_name, last_name, email, replace(phone_number, '.', '-') phone_number, hire_date, salary, commission_pct, department_id
from employees;

6. Сколько сотрудников которые работают в одном и том же отделе и получают
одинаковую зарплату?

select department_id, salary, count(*) number_of_employees from employees
	group by department_id, salary
	order by number_of_employees desc;

7. Получить список сотрудников с самым длинным именем

select first_name from employees
	where length(first_name) = (select MAX(length(first_name)) from employees);
